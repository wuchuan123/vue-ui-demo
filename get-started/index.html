# 快速上手

# 引入必要资源

import {Button} from "CC-ui-demo";
import "CC-ui-demo/dist/index.css";

# 注册组件

export default {
  name: "ComponentName",
  components: {
    "i-button": Button,
  },
}

# 使用

代码：

<template>
  <div>
    <i-button>默认按钮</i-button>
  </div>
</template>
